#.*	/* comments */
(?x: [[:digit:]]+
   | 0b[01]+
   | 0x[[:xdigit:]])	{ return INT; }
(?x: [[:digit:]]+f
   | [[:digit:]]+\.[[:digit:]]*
   | [[:digit:]]*\.[[:digit:]]+)	{ return REAL; }
\"(\\\"|[^"])*\"	{ return STRING; }
(?x: true
   | false)	{ return BOOL; }
if	{ return IF; }
else	{ return ELSE; }
for	{ return FOR; }
forall	{ return FORALL; }
while	{ return WHILE; }
def	{ return DEF; }
PATTERNHERE	{ return INDENT; }
PATTERNHERE	{ return DEDENT; }
<	{ return LT; }
>	{ return GT; }
<=	{ return LTE; }
>=	{ return GTE; }
==	{ return EQ; }
!=	{ return NEQ; }
+	{ return PLUS; }
-	{ return MINUS; }
/	{ return DIV; }
*	{ return MULT; }
**	{ return EXP; }
/=	{ return DIVEQ; }
*=	{ return MULTEQ; }
+=	{ return PLUSEQ; }
-=	{ return MINUSEQ; }
&&	{ return AND; }
\|\|	{ return OR; }
[[:alpha:]_][[:alnum:]_]*	{ return IDENT; }
